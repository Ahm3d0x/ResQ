create table public.ambulances (
  id bigint generated by default as identity not null,
  code character varying(100) not null,
  lat numeric(10, 7) null,
  lng numeric(10, 7) null,
  status public.ambulance_status_enum not null default 'available'::ambulance_status_enum,
  created_at timestamp with time zone not null default CURRENT_TIMESTAMP,
  updated_at timestamp with time zone not null default CURRENT_TIMESTAMP,
  driver_id bigint null,
  constraint ambulances_pkey primary key (id),
  constraint ambulances_code_key unique (code),
  constraint ambulances_driver_id_fkey foreign KEY (driver_id) references users (id) on delete set null
) TABLESPACE pg_default;

create trigger update_ambulances_updated_at BEFORE
update on ambulances for EACH row
execute FUNCTION update_updated_at_column ();

create table public.hospital_beds (
  id bigint generated by default as identity not null,
  hospital_id bigint not null,
  bed_number integer not null default 1,
  status character varying(20) not null default 'available'::character varying,
  patient_name character varying(255) null,
  patient_id character varying(255) null,
  incident_id bigint null,
  admission_time timestamp with time zone null,
  created_at timestamp with time zone not null default CURRENT_TIMESTAMP,
  constraint hospital_beds_pkey primary key (id),
  constraint hospital_beds_hospital_id_fkey foreign KEY (hospital_id) references hospitals (id) on update RESTRICT on delete CASCADE,
  constraint hospital_beds_incident_id_fkey foreign KEY (incident_id) references incidents (id) on update RESTRICT on delete set null,
  constraint hospital_beds_status_check check (
    (
      (status)::text = any (
        (
          array[
            'available'::character varying,
            'occupied'::character varying,
            'blocked'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_hospital_beds_hospital on public.hospital_beds using btree (hospital_id) TABLESPACE pg_default;

create index IF not exists idx_hospital_beds_status on public.hospital_beds using btree (status) TABLESPACE pg_default;
